69 {

71 
72  std::vector<V*> vecs(m_sets.size(),(V*) NULL);
73  for (unsigned int i = 0; i < vecs.size(); ++i) {
74  vecs[i] = new V(m_sets[i]->vectorSpace().zeroVector());

77  unsigned int cummulativeSize = 0;
